export function Counter() {
  // We can use a simple global-style variable directly in MDX for the demo!
  // This avoids the Preact hooks async import error entirely.
  return (
    <div style={{ 
      padding: '1rem', 
      background: '#f4f4f5', 
      borderRadius: '8px', 
      marginTop: '1rem',
      display: 'inline-block'
    }}>
      <strong>Live Component: </strong>
      <button 
        onClick={(e) => {
          const btn = e.target;
          const count = parseInt(btn.getAttribute('data-count') || '0') + 1;
          btn.setAttribute('data-count', count);
          btn.innerText = `Clicked ${count} times`;
        }}
        style={{ padding: '0.2rem 0.6rem', cursor: 'pointer' }}
      >
        Clicked 0 times
      </button>
    </div>
  );
}
